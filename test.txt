boolean isOdd(integer number){
    return number%2;
}

boolean isEven(integer number){
    return !number%2;
}

integer main(){
    integer number = 19;
    if (isEven(number)) {
        if (!isOdd(number)){
            if (number<=50){
                number=88;
                number=number/number;
            }
            else {
                number=number*number;
            }
        }
        else{
            number=15;
        }
    }

    return 0;
}
